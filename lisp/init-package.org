*** check my installed package state
    #+begin_src emacs-lisp
      (require 'cl)
      (defun revolt/packages-installed-p()
	(loop for pkg in revolt/package-list
	      when (not (package-installed-p pkg))
	      do (return nil)
	      finally (return t)
	      ))
    #+end_src
